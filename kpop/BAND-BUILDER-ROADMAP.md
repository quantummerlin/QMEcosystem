### Backend/Calculation Logic

**Synergy Calculator (synergy-calculator.js):**
```javascript
function calculateGroupSynergy(members) {
  // 1. Calculate pairwise compatibility (all combinations)
  let pairScores = [];
  for (let i = 0; i < members.length; i++) {
    for (let j = i + 1; j < members.length; j++) {
      let score = getCompatibility(members[i], members[j]);
      pairScores.push(score);
    }
  }
  
  // 2. Average pairwise scores
  let avgPairScore = pairScores.reduce((a,b) => a+b) / pairScores.length;
  
  // 3. Check role balance (bonus points)
  let roleBalance = calculateRoleBalance(members);
  
  // 4. Debut date alignment (if provided)
  let debutBonus = 0;
  if (debutDate) {
    debutBonus = calculateDebutAlignment(members, debutDate);
  }
  
  // Final score
  return Math.min(100, avgPairScore * 0.7 + roleBalance * 0.2 + debutBonus * 0.1);
}
```

**Role Assignment Logic:**
```javascript
function assignRoles(members) {
  let roles = {
    leader: null,
    mainVocal: null,
    mainRapper: null,
    mainDancer: null,
    visual: null,
    center: null,
    maknae: null
  };
  
  // Leader: Highest Life Path or Expression number
  roles.leader = members.reduce((prev, curr) => 
    curr.lifePath > prev.lifePath ? curr : prev
  );
  
  // Main Vocal: Highest Soul Urge (emotional expression)
  roles.mainVocal = members.reduce((prev, curr) => 
    curr.soulUrge > prev.soulUrge ? curr : prev
  );
  
  // Main Rapper: Mars or Mercury dominant
  roles.mainRapper = members.find(m => 
    m.mercurySign === 'Gemini' || m.marsSign === 'Aries'
  );
  
  // Visual: Venus dominant + high Personality number
  roles.visual = members.reduce((prev, curr) => 
    curr.personality > prev.personality ? curr : prev
  );
  
  // Maknae: Youngest member
  roles.maknae = members.reduce((prev, curr) => 
    new Date(curr.dob) > new Date(prev.dob) ? curr : prev
  );
  
  return roles;
}
```

**Group Name Generator:**
```javascript
function generateGroupNames(members, count = 5) {
  let names = [];
  
  // Method 1: Initials
  let initials = members.map(m => m.name[0]).join('');
  names.push(initials.toUpperCase());
  
  // Method 2: Numerology sum
  let numSum = members.reduce((sum, m) => sum + m.lifePath, 0) % 9 || 9;
  let numWords = ['Nova', 'Pulse', 'Eclipse', 'Aura', 'Vibe', 'Starlight', 'Phoenix', 'Cosmos', 'Zenith'];
  names.push(numWords[numSum - 1]);
  
  // Method 3: K-pop style combinations
  let prefixes = ['Dream', 'Star', 'Super', 'Cosmic', 'Quantum', 'Crystal', 'Lunar', 'Solar'];
  let suffixes = ['Squad', 'Wave', 'Nova', 'Kids', 'Girls', 'Boys', 'Crew', 'Force', 'X'];
  names.push(randomChoice(prefixes) + randomChoice(suffixes));
  
  // Method 4: Korean syllable fusion
  let koreanSyllables = ['Ha', 'Na', 'Ra', 'Mi', 'So', 'Bi', 'Cha', 'Ka', 'Ta', 'Pa'];
  names.push(randomChoice(koreanSyllables) + randomChoice(koreanSyllables) + randomChoice(koreanSyllables));
  
  // Method 5: Mystical/Cosmic
  let mystical = ['Ethereal', 'Celestial', 'Nebula', 'Astral', 'Radiant', 'Luminous', 'Infinite', 'Stellar'];
  names.push(randomChoice(mystical));
  
  return names.slice(0, count);
}
```
