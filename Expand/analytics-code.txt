  gtag('config', 'G-XXXXXXXXXX');
  
  // Custom event tracking for Quantum Merlin
  function trackEvent(eventName, parameters) {

  function trackToolUsage(toolName, action) {

      tool_name: toolName,
      action: action,

    });
  }
  
  // Jukebox tracking
  function trackJukeboxAction(action, itemType) {
    gtag('event', 'jukebox_action', {

    });
  }
