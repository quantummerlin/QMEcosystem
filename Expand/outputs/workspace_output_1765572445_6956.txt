    /**
     * Get tracks by category
     */
    getTracksByCategory(categoryId) {
        return this.tracks.filter(track => track.category === categoryId);
    },
    
    /**
     * Get featured tracks
     */
    getFeaturedTracks() {
        return this.tracks.filter(track => track.featured === true);
    },
    
    /**
     * Get recommended tracks
     */
    getRecommendedTracks() {
        return this.tracks.filter(track => track.recommended === true);
    },
    
    /**
     * Search tracks by name or tags
     */
    searchTracks(query) {
        const lowerQuery = query.toLowerCase();
        return this.tracks.filter(track => 
            track.name.toLowerCase().includes(lowerQuery) ||
            track.description.toLowerCase().includes(lowerQuery) ||
            track.tags.some(tag => tag.includes(lowerQuery))
        );
    },
    
    /**
     * Get track by ID
     */
    getTrackById(id) {
        return this.tracks.find(track => track.id === id);
    },
    
    /**
     * Get top tracks by play count
     */
    getTopTracks(limit = 10) {
        return [...this.tracks]
            .sort((a, b) => b.playCount - a.playCount)
            .slice(0, limit);
    },
    
    /**
     * Get tracks by tag
     */
    getTracksByTag(tag) {
        return this.tracks.filter(track => track.tags.includes(tag));
    },
    
    /**
     * Get all unique tags
     */
    getAllTags() {
        const tags = new Set();
        this.tracks.forEach(track => {
            track.tags.forEach(tag => tags.add(tag));
        });
        return Array.from(tags).sort();
    }
};
