        // Update week overview
        function updateWeekOverview() {
            const weekGrid = document.getElementById('weekGrid');
            weekGrid.innerHTML = '';
            
            const today = new Date();
            const weekDays = ['Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday'];
            
            for (let i = 0; i < 7; i++) {
                const date = new Date(today);
                date.setDate(today.getDate() - today.getDay() + i);
                
                const dayOfYear = Math.floor((date - new Date(date.getFullYear(), 0, 0)) / 86400000);
                const elementIndex = dayOfYear % 4;
                const elementOfDay = elements[elementIndex];
                const frequencyOfDay = ((dayOfYear + date.getFullYear()) % 9) || 9;
                
                const dayCard = document.createElement('div');
                dayCard.className = 'day-card';
                
                dayCard.innerHTML = `
                    <div class="day-name">${weekDays[i]}</div>
                    <div class="day-element">${elementOfDay}</div>
                    <div class="day-frequency">${frequencyOfDay} Hz</div>
                `;
                
                weekGrid.appendChild(dayCard);
            }
        }
